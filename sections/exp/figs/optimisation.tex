\newcommand{\optnode}[5]{
    % supernode id, find content, assume content, found content, position 
    \node (#1-find) [smallamber,minimum height=0.1cm ,text width=1.8cm,minimum width = 2cm #5] {\specialcell{#2}};
    \node (#1-assume) [smalldarkbyzantium,minimum height=0.1cm, text width=1.8cm, minimum width = 2cm, below=0.15cm of #1-find] {\specialcell{#3}};
    \node (#1-found) [smallazure,minimum height=0.1cm, text width=1.8cm, minimum width = 2cm, below=0.15cm of #1-assume] {\specialcell{#4}};
    \begin{pgfonlayer}{background}
        \node(#1)[bigblush] [fit = (#1-find) (#1-found)] {};
    \end{pgfonlayer}
}
